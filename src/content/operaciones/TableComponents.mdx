export function OperationTable() {
  // Función para extraer el mes a partir de la fecha (formato DD/MM/YYYY)
  const getMonthYear = (dateStr) => {
    const parts = dateStr.split('/');
    if (parts.length === 3) {
      const month = parts[1];
      const year = parts[2];
      return `${month}/${year}`;
    }
    return dateStr; // Retorna la fecha original si no se puede parsear
  };

  // Agrupar operaciones por mes
  const groupedOperations = [];
  let currentMonth = null;

  operations.forEach((operation) => {
    const operationMonth = getMonthYear(operation.date);
    
    // Si cambiamos de mes, añadimos un separador
    if (operationMonth !== currentMonth) {
      currentMonth = operationMonth;
      
      // Convertir el formato de mes/año a un formato más legible
      let monthName;
      switch (currentMonth.split('/')[0]) {
        case '01': monthName = 'ENERO'; break;
        case '02': monthName = 'FEBRERO'; break;
        case '03': monthName = 'MARZO'; break;
        case '04': monthName = 'ABRIL'; break;
        case '05': monthName = 'MAYO'; break;
        case '06': monthName = 'JUNIO'; break;
        case '07': monthName = 'JULIO'; break;
        case '08': monthName = 'AGOSTO'; break;
        case '09': monthName = 'SEPTIEMBRE'; break;
        case '10': monthName = 'OCTUBRE'; break;
        case '11': monthName = 'NOVIEMBRE'; break;
        case '12': monthName = 'DICIEMBRE'; break;
        default: monthName = currentMonth;
      }
      
      // Añadir un objeto especial para el separador de mes
      groupedOperations.push({
        isMonthlySeparator: true,
        monthLabel: `${monthName} ${currentMonth.split('/')[1]}`
      });
    }
    
    // Añadir la operación actual
    groupedOperations.push(operation);
  });

  return (
    <div class="flex justify-center -mt-6">
      <table class="w-full border-separate border-zinc-950 border-2 rounded">
        <caption class="caption-top">
          <h1 class="text-3xl font-bold text-center tracking-wide">Ordenes realizadas</h1>
        </caption>
        <caption class="caption-bottom">
          <p class="text-[11px] sm:text-base"><a href="#colors">📌 Explicación de los colores</a></p>
          <p class="text-yellow-500 text-[11px] sm:text-base">* El color amarillo representa la protección de diciembre frente a una desviación de la estrategia que se está siguiendo. Se cubre de esta manera una posible bajada aún mayor de la dominancia de BTC o en su defecto una alta subida de BTC.</p>
        </caption>
        <thead class="bg-gray-100">
          <tr>
            <th class="c-titles">FECHA</th>
            <th class="c-titles">TOKEN</th>
            <th class="c-titles">CANTIDAD</th>
            <th class="c-titles">PAR</th>
            <th class="c-titles">OPERACIÓN</th>
            <th class="c-titles">Precio y moneda</th>
          </tr>
        </thead>
        <tbody>
          {groupedOperations.map((item, index) => {
            // Si es un separador de mes, renderizamos una fila especial
            if (item.isMonthlySeparator) {
              return (
                <tr key={`month-${index}`} class="bg-gray-100">
                  <td colSpan="6" class="text-center font-bold py-2 border-b border-t border-zinc-950">
                    {item.monthLabel}
                  </td>
                </tr>
              );
            }
            
            // Si es una operación normal, renderizamos como antes
            const { date, token, amount, pair, operation, price, details, color } = item;
            return (
              <tr key={`op-${index}`}>
                <td class="c-normal c-underline">{date}</td>
                <td class={`${color}`}>
                  {details ? (
                    <details class="group">
                      <summary class="cursor-pointer">{token}</summary>
                      <p class="mt-2 text-[9px] sm:text-base text-gray-600">{details}</p>
                    </details>
                  ) : (
                    token
                  )}
                </td>
                <td class="c-normal">{amount}</td>
                <td class="c-normal">{pair}</td>
                <td class={`c-normal ${operation}`}></td>
                <td class="c-normal-left">{price}</td>
              </tr>
            );
          })}
        </tbody>
      </table>
    </div>
  );
}
